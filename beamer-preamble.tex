\begin{center}
  \setlength{\textwidth}{#1}
  \begin{minipage}{#1}
  \begin{actionenv}#3%
    \def\insertblocktitle{#2}%
    \par%
    \usebeamertemplate{block begin} \vspace{0.5em}}
  {\par%
    \usebeamertemplate{block end}%
  \end{actionenv}
  \end{minipage}
\end{center}
\vspace{1em}}
