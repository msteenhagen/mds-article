%% # Article-class preamble  
%% 
%% Mark Sprevak (<http://sites.google.com/site/msprevak>)
%% 
%% * Examples of typeset PDFs: <http://goo.gl/Or8cC>  
%% * A minimal working example: <http://goo.gl/fDLrq>
%% 
%% # Usage
%% 
%% This file should be `\input{}` in first line of a .tex file, e.g.:
%% 
%%     \input{article-preamble.tex} 
%%     \title{My article title}
%%     \published{Forthcoming in the \emph{The Journal}}
%%     \doi{10.1001/abcd/abc001}
%% 
%%     \begin{document}
%%     ...
%%     \end{document}
%% 
%%
%% # Useful commands
%% 
%% `\title{...}`:			document title  
%% `\date{...}`:			document date---if absent: use `\gitAuthorDate`  
%% `\published{...}`:	publication info---if absent: show 'Draft only'  
%% `\doi{...}`:				document DOI  
%% `\headers{}`:			suppress running headers (useful for short documents)  
%% 
%% # Prerequisites
%% 
%% 1. A modern LaTeX installation
%% 2. Minion Pro fonts: <http://goo.gl/lQqMy>
%% 3. Biblatex
%% 4. The `gitinfo` package
%% 
%% # Acknowledgements
%% 
%% Borrows from many sources, most notably Kieran Healy's design:
%% <https://github.com/kjhealy/latex-custom-kjh>
 
\documentclass[oneside,12pt]{article}
\usepackage{fixltx2e}
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage[a4paper,showframe]{geometry}
% \usepackage[a4paper]{geometry}
%% MinionPro: remove `opticals` and `fullfamily` if you don't have full font
\usepackage[opticals,fullfamily,mathlf,minionint,footnotefigures]{MinionPro}
\usepackage[scaled=0.85]{inconsolata}
\usepackage{booktabs}
\usepackage{graphicx}
\usepackage{xcolor}
\usepackage{enumitem}
\usepackage{titling}
\usepackage{gitinfo}
\usepackage[small,topmarks,pagestyles]{titlesec}
\usepackage[british,cleanlook]{isodate}
\usepackage[british]{babel}
\usepackage[autostyle]{csquotes}
\usepackage[babel]{microtype}


\def\myauthor{Mark Sprevak}
\def\myaffiliation{University of Edinburgh}
\def\myemail{\texttt{\href{mailto:mark.sprevak@ed.ac.uk}{mark.sprevak@ed.ac.uk}}}

\geometry{
	top=2.5cm,
	bottom=3.5cm,
	headheight=3ex,
	headsep=2ex
}

%% Bibliography - biblatex 
%%
\usepackage[useprefix,authordate,natbib,sortcites,backend=biber]{biblatex-chicago}
\setlength\bibitemsep{6pt}
\defbibheading{bibliography}[\refname]{\section*{#1}}
%% addbibresource: change `refs.bib` to the name of your .bib file
\addbibresource{refs.bib}


%% Header: title page
%%
%% Center footer:	page number
%% Right header:	email address
%% Left header:		publication info/'Draft only', DOI if \doi{..} set
\newcommand{\published}[1]{\def\publicationinfo{#1}}
\newcommand{\doi}[1]{\def\doiinfo{#1}}
\newcommand{\headers}[1]{\def\setnoheaders{#1}}

\renewpagestyle{plain}{
	\setfoot{}{\thepage}{}
	\ifdefined\publicationinfo
		\sethead{{\parbox[t][3cm]{.70\textwidth}
						 {\footnotesize \strut \publicationinfo \doiinsert}}}
						{}
						{\footnotesize \myemail}
	\else
		\sethead{\footnotesize \strut 
						 \emph{Draft only. Please do not cite without permission.}}
						{}
						{\footnotesize \strut \myemail}
 	\fi
}

%% Header: running headers
%%
%% Center footer:	page number
%% Right header:	subsection title
%% Left header:		section number and title
\newpagestyle{fancy}{
	\headrule
	\setfoot{}{\thepage}{}
	\sethead{\small \thesection \quad \sectiontitle}{}{\small \subsectiontitle}
	% \sethead{test}{}{}
}
\pagestyle{fancy}

%% Description labels: bold, small caps, and lowercase
%% Description: forced on next line if label is too long
\renewcommand{\descriptionlabel}[1]{\hspace*{\labelsep}\textsc{\textbf{\MakeLowercase{#1}}}}
\setlist[description]{style=nextline}

%% Section titles: sub-subsection headings italic
\titleformat{\subsubsection}
{\itshape\normalsize}{\thesubsubsection}{1em}{}

%% URL font: matches surrounding text
% \urlstyle{same}

%% Dark purple for hyperlinks
\definecolor{purple_link}{RGB}{76, 64, 133}

%% Footnotes: stay on the same page
\interfootnotelinepenalty=10000 	

%% Text can stretch a little into margins to avoid bad line-breaks
\setlength{\emergencystretch}{3em}  

\usepackage[
	plainpages = false,
	pdfpagelabels,
	bookmarksnumbered,
	pdftex,
	colorlinks = true,
	citecolor  = purple_link,
	linkcolor  = purple_link,
	urlcolor   = purple_link
]{hyperref}

%% Quote environment: take up less space (copied from savetrees.sty)
\makeatletter
\renewenvironment{quote}
               {\list{}{\listparindent=0pt%whatever you need
                        \itemindent    \listparindent
                        \leftmargin=2em%  whatever you need
                        \rightmargin=2em%whatever you need
                        \topsep=0pt%%%%%  whatever you need
                        \parsep        \z@ \@plus\p@}%
                \item\relax}
               {\endlist}
\makeatother

%% aquote environment: citation flush right, newline when necessary
%% 	usage: begin{aquote}{\citep{Lewis00}} ... \end{aquote} 
\def\signed #1{{\leavevmode\unskip\nobreak\hfil\penalty50\hskip2em
  \hbox{}\nobreak\hfil#1%
  \parfillskip=0pt \finalhyphendemerits=0 \endgraf}}

\newsavebox\mybox
\newenvironment{aquote}[1]
  {\savebox\mybox{#1}\begin{quote}}
  {\signed{\usebox\mybox}\end{quote}}

%% Title, author, and institution
\pretitle{\par\vskip 3em \begin{flushleft}\LARGE\bfseries}
\posttitle{\end{flushleft}\par\vskip 0.5em}
\preauthor{\begin{flushleft}\normalsize\bigskip}
\postauthor{\end{flushleft}}
\predate{\begin{flushleft}\normalsize}
\postdate{\end{flushleft}}

\author{\myauthor 					\\
  			\emph{\myaffiliation}}

%% Date
%% \gitAuthorDate set: set that as the date
%% Otherwise: set '(None)' as the date
%% This is overridden if \date{...} set manually in document
\ifnum\pdfstrcmp{\gitAuthorDate}{(None)}=0 
	\date{\gitAuthorDate}
\else
	\date{\printdate{\gitAuthorDate}}
\fi

%% Abstract: plain quote environment
\renewenvironment{abstract}{\begin{quote}}{\end{quote}}

%% Inserted after \begin{document}
\AtBeginDocument{

	%% Metadata in output PDF: title and author
	\hypersetup{
		pdftitle = {\thetitle},
		pdfauthor = {\myauthor}
	}

	\ifdefined\doiinfo
		\newcommand{\doilink}[1]{\par\texttt{\href{http://dx.doi.org/#1}{doi:#1}}}
		\newcommand{\doiinsert}{\doilink{\doiinfo}}
	\else
		\newcommand{\doiinsert}{\relax}
	\fi
	
	\maketitle
	
	%% \headers{} suppresses running headers
	\ifdefined\setnoheaders
		\sethead{}{}{}
		\setheadrule{0pt}
	\fi
	
	%% Paragraphs: don't indent first line
	%% Paragraphs: add small space to separate paragraphs
	\setlength{\parindent}{0pt}
	\setlength{\parskip}{6pt plus 2pt minus 1pt}
}

%% Inserted before \end{document}
\AtEndDocument{

	% Header: running headers for References
	% \headers{} suppresses running headers
	\ifdefined\setnoheaders
		\sethead{}{}{}
		\setheadrule{0pt}
	\else
		\sethead{\small \refname}{}{}
	\fi
		
	%% Bibliography - biblatex
	\printbibliography
	
	%% Git date and commit hash: print if set, otherwise do nothing
	\ifnum\pdfstrcmp{\gitAuthorDate}{(None)}=0 
		\relax 
	\else
		\flushright{\sffamily \scriptsize \gitAbbrevHash\ on \gitAuthorDate}
	\fi
	
}
